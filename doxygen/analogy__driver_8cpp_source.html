33 DAQ::Device *AnalogyDriver::createDevice(const std::list<std::string> &args) {
34  int err = 0;
35  a4l_desc_t dsc;
36  a4l_sbinfo_t *sbinfo;
37  std::string name = args.front();
38 
39  err = a4l_open(&dsc, name.c_str());
40  if(err < 0) {
41  ERROR_MSG("AnalogyDriver::createDevice : unable to open %s (err=%d).\n", name.c_str(), err);
42  return 0;
43  }
44 
45  // Allocate a buffer to get more info (subd, chan, rng)
46  dsc.sbdata = malloc(dsc.sbsize);
47  if (dsc.sbdata == NULL) {
48  err = -ENOMEM;
49  ERROR_MSG("AnalogyDriver: info buffer allocation failed\n");
50  return 0;
51  }
52 
53  // Get this data
54  err = a4l_fill_desc(&dsc);
55  if (err < 0) {
56  ERROR_MSG("AnalogyDriver: a4l_fill_desc failed (err=%d)\nPlease run rtxi_load_analogy script in scripts directory.\n", err);
57  return 0;
58  }
59 
60  // We need to find each subdevice index manually since idx_*_subd often fails
61  // Go over all subdevices and save the indexes of the first AI, AO and DIO
62  int idx_ai = -1;
63  int idx_ao = -1;
64  int idx_dio = -1;
65  for (int i=0; i < dsc.nb_subd; i++) {
66  err = a4l_get_subdinfo(&dsc, i, &sbinfo);
67  if(err != 0) {
68  ERROR_MSG("AnalogyDriver: a4l_get_subd_info failed, wrong subdevice index %i (err=%d)\n",
69  err, i);
70  return 0;
71  }
72  // Assign index; save just the first device if many
73  if (((sbinfo->flags & A4L_SUBD_TYPES )== A4L_SUBD_AI) && (idx_ai < 0))
74  idx_ai = i;
75  else if (((sbinfo->flags & A4L_SUBD_TYPES) == A4L_SUBD_AO) && (idx_ao < 0))
76  idx_ao = i;
77  else if (((sbinfo->flags & A4L_SUBD_TYPES) == A4L_SUBD_DIO) && (idx_dio < 0))
78  idx_dio = i;
79  }
80 
81  size_t count[5] = { 0, 0, 0, 0, 0,};
82  err = a4l_get_subdinfo(&dsc, idx_ai, &sbinfo);
83  if(err == 0)
84  count[0] = sbinfo->nb_chan;
85  err = a4l_get_subdinfo(&dsc, idx_ao, &sbinfo);
86  if(err == 0)
87  count[1] = sbinfo->nb_chan;
88  err = a4l_get_subdinfo(&dsc, idx_dio, &sbinfo);
89  if(err == 0)
90  count[2] = sbinfo->nb_chan;
91 
92  if(!(count[0]+count[1]+count[2]+count[3]+count[4])) {
93  ERROR_MSG("AnalogyDriver::createDevice : no Analogy device configured on %s.\n",name.c_str());
94  a4l_close(&dsc);
95  return 0;
96  }
97 
98  IO::channel_t channel[count[0]+count[1]+2*count[2]];
99  for(size_t i=0;i<count[0];++i) {
100  std::ostringstream name;
101  name << "Analog Input " << i;
102  channel[i].name = name.str();
103  channel[i].description = "";
104  channel[i].flags = IO::OUTPUT;
105  }
106  for(size_t i=count[0];i<count[0]+count[1];++i) {
107  std::ostringstream name;
108  name << "Analog Output " << i-count[0];
109  channel[i].name = name.str();
110  channel[i].description = "";
111  channel[i].flags = IO::INPUT;
112  }
113  for(size_t i=count[0]+count[1];i<count[0]+count[1]+count[2];++i) {
114  std::ostringstream name;
115  name << "Digital I/O " << i-count[0]-count[1];
116  channel[i].name = name.str();
117  channel[i].description = "";
118  channel[i].flags = IO::OUTPUT;
119  }
120  for(size_t i=count[0]+count[1]+count[2];i<count[0]+count[1]+2*count[2];++i) {
121  std::ostringstream name;
122  name << "Digital I/O " << i-count[0]-count[1]-count[2];
123  channel[i].name = name.str();
124  channel[i].description = "";
125  channel[i].flags = IO::INPUT;
126  }
127  for(size_t i=count[0]+count[1]+2*count[2];i<count[0]+count[1]+2*count[2]+count[3];++i) {
128  std::ostringstream name;
129  name << "Digital Input " << i-count[0]-count[1]-2*count[2];
130  channel[i].name = name.str();
131  channel[i].description = "";
132  channel[i].flags = IO::OUTPUT;
133  }
134  for(size_t i=count[0]+count[1]+2*count[2]+count[3];i<count[0]+count[1]+2*count[2]+count[3]+count[4];++i) {
135  std::ostringstream name;
136  name << "Digital Output " << i-count[0]-count[1]-2*count[2]-count[3];
137  channel[i].name = name.str();
138  channel[i].description = "";
139  channel[i].flags = IO::INPUT;
140  }
141 
142  AnalogyDevice *dev = new AnalogyDevice(&dsc,name,channel,count[0]+count[1]+2*count[2]);
143  devices.push_back(dev);
144  return dev;
145 
146  return 0;
147 }
