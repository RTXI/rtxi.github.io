234 // Scales data based upon desired settings for the channel
235 void Scope::setData(double data[],size_t size) {
236  if(isPaused)
237  return;
238 
239  if(size < getChannelCount()) {
240  ERROR_MSG("Scope::setData() : data size mismatch detected\n");
241  return;
242  }
243 
244  size_t index = 0;
245  for(std::list<Channel>::iterator i = channels.begin(), end = channels.end();i != end;++i) {
246  i->data[data_idx] = data[index++];
247 
248  if(triggering && i == triggerChannel &&
249  ((triggerDirection == POS && i->data[data_idx-1] < triggerThreshold && i->data[data_idx] > triggerThreshold) ||
250  (triggerDirection == NEG && i->data[data_idx-1] > triggerThreshold && i->data[data_idx] < triggerThreshold))) {
251  triggerQueue.push_back(data_idx);
252  }
253  }
254 
255  ++data_idx %= data_size;
256 
257  if(isPaused || getChannelCount() == 0)
258  return;
259 
260  if(triggering && !triggerQueue.empty() && (data_idx+2)%data_size == triggerQueue.front()) {
261  if(triggerLast != (size_t)(-1) && (triggerQueue.front()+data_size-triggerLast)%data_size*period < triggerHoldoff)
262  triggerQueue.pop_front();
263  else {
264  triggerLast = triggerQueue.front();
265  triggerQueue.pop_front();
266 
267  if(!triggerHolding)
268  drawCurves();
269 
270  for(std::list<Channel>::iterator i = channels.begin(), iend = channels.end();i != iend;++i) {
271  std::vector<double> x (i->data.size());
272  std::vector<double> y (i->data.size());
273  double *x_loc = x.data();
274  double *y_loc = y.data();
275 
276  // Set scale map for channel
277  scaleMapX->setScaleInterval(0, hScl*divX);
278  scaleMapY->setScaleInterval(-i->scale*divY/2, i->scale*divY/2);
279 
280  // Scale data to pixel coordinates
281  for(size_t j = 0; j < i->data.size(); ++j) {
282  *x_loc = scaleMapX->transform(j*period);
283  *y_loc = scaleMapY->transform(i->data[(data_idx+j)%i->data.size()]+i->offset);
284  ++x_loc;
285  ++y_loc;
286  }
287  // Append data to curve
288  // Makes deep copy - which is not optimal
289  // TODO: change to pointer based method
290  i->curve->setSamples(x.data(), y.data(), i->data.size());
291  }
292  }
293  }
294 }

331 void Scope::setTrigger(trig_t direction,double threshold,std::list<Channel>::iterator channel,bool holding,double holdoff) {
332  triggerHolding = holding;
333  triggerHoldoff = holdoff;
334  triggerLast = (size_t)(-1);
335 
336  if(triggerChannel != channel || triggerThreshold != threshold) {
337  triggerChannel = channel;
338  triggerThreshold = threshold;
339  }
340 
341  if(triggerDirection != direction) {
342  if(direction == NONE) {
343  triggering = false;
344  timer->start(refresh);
345  triggerQueue.clear();
346  } else {
347  triggering = true;
348  timer->stop();
349  }
350  triggerDirection = direction;
