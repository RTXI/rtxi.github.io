220 // Scales data based upon desired settings for the channel
221 void Scope::setData(double data[],size_t size) {
222  if(isPaused)
223  return;
224 
225  if(size < getChannelCount()) {
226  ERROR_MSG("Scope::setData() : data size mismatch detected\n");
227  return;
228  }
229 
230  size_t index = 0;
231  for(std::list<Channel>::iterator i = channels.begin(), end = channels.end();i != end;++i) {
232  i->data[data_idx] = data[index++];
233 
234  if(triggering && i == triggerChannel &&
235  ((triggerDirection == POS && i->data[data_idx-1] < triggerThreshold && i->data[data_idx] > triggerThreshold) ||
236  (triggerDirection == NEG && i->data[data_idx-1] > triggerThreshold && i->data[data_idx] < triggerThreshold))) {
237  triggerQueue.push_back(data_idx);
238  }
239  }
240 
241  ++data_idx %= data_size;
242 
243  if(triggering && !triggerQueue.empty() && (data_idx+2)%data_size == triggerQueue.front()) {
244  if(triggerLast != (size_t)(-1) && (triggerQueue.front()+data_size-triggerLast)%data_size*period < triggerHoldoff)
245  triggerQueue.pop_front();
246  else {
247  triggerLast = triggerQueue.front();
248  triggerQueue.pop_front();
249 
250  if(!triggerHolding)
251  drawCurves();
252 
253  for(std::list<Channel>::iterator i = channels.begin(), iend = channels.end();i != iend;++i) {
254  std::vector<double> x (i->data.size());
255  std::vector<double> y (i->data.size());
256  double *x_loc = x.data();
257  double *y_loc = y.data();
258 
259  // Scale data to pixel coordinates
260  for(size_t j = 0; j < i->data.size(); ++j) {
261  *x_loc = ((j*period)*width())/(hScl*divX);
262  *y_loc = i->data[(data_idx+j)%i->data.size()];
263  ++x_loc;
264  ++y_loc;
265  }
266  // Plot
267  i->curve->setRawSamples(x.data(), y.data(), (int)i->data.size());
268  }
269  }
270  }
271 }

307 void Scope::setTrigger(trig_t direction,double threshold,std::list<Channel>::iterator channel,bool holding,double holdoff) {
308  triggerHolding = holding;
309  triggerHoldoff = holdoff;
310  triggerLast = (size_t)(-1);
311 
312  if(triggerChannel != channel || triggerThreshold != threshold) {
313  triggerChannel = channel;
