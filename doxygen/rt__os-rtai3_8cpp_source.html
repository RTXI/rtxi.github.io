65 int RT::OS::initiate(void) {
66  if (rtai_lxrt_loaded()) {
67  ERROR_MSG("RTOS:RTAI::initiate : rtai_lxrt module not loaded.\n");
68  return -ENOENT;
69  }
70 
71  rt_allow_nonroot_hrt();
72  rt_set_oneshot_mode();
73  start_rt_timer(0);

75  /**************************************************************************************
76  * On some systems like Fedora Core 4 the memory footprint for rtxi exceeds the mlock *
77  * limit allowed by the linux kernel. As root this limit can be removed with a call *
78  * to setrlimit. However, as a non-root user the call to setrlimit will fail. So if *
79  * you aren't root setrlimit will fail, but that is okay if your system is one that *
80  * doesn't need the extra space. *
81  **************************************************************************************/
82 
83  struct rlimit rlim = { RLIM_INFINITY, RLIM_INFINITY };
84  setrlimit(RLIMIT_MEMLOCK,&rlim);
85 
86  if (mlockall(MCL_CURRENT | MCL_FUTURE)) {
87  stop_rt_timer();
88  ERROR_MSG("RTOS:RTAI::initiate : failed to lock memory.\n");
89  return -EPERM;
90  }
91 
92  pthread_key_create(&is_rt_key,0);
93  init_rt = true;
94 
95  return 0;
96 }

111 static void *bounce(void *bounce_info) {
112  rtai3_bounce_info_t *info = reinterpret_cast<rtai3_bounce_info_t *>(bounce_info);
113 
114  rtai3_task_t *t = info->t;
115  void *(*entry)(void *) = info->entry;
116  void *arg = info->arg;
117 
118  if (!(t->task = rt_task_init_schmod(reinterpret_cast<unsigned long>(t),1,2000,0,SCHED_FIFO,info->prio))) {
119  ERROR_MSG("RT::OS::createTask : failed to create task\n");
120  return reinterpret_cast<void *>(-EPERM);
121  }
122 
123  t->period = -1;
124  t->next_t = -1;
125  t->wakeup = -1;
126  t->thread = pthread_self();
127 
128  pthread_setspecific(is_rt_key,reinterpret_cast<const void *>(t));
129 
130  rt_make_hard_real_time();
131  rtai_save_flags_and_cli(t->eflags);
132 
133  sem_post(&info->sem);
134 
135  void *retval = entry(arg);
136 
137  rtai_restore_flags(t->eflags);
138  rt_task_delete(t->task);
139 
140  return retval;
141 }

181 int RT::OS::setPeriod(RT::OS::Task task,long long period) {
182  rtai3_task_t *t = reinterpret_cast<rtai3_task_t *>(task);
183 
184  /*
185  * Set how early the realtime thread should wakeup.
186  * Upper limit constraints:
187  * It doesn't make sense for this value to exceed the period.
188  * The earlier the realtime thread wakes up the less time it spends in a blocking sleep,
189  * which means less time for the non-realtime thread to work. Thus the machine could
190  * appear crashed if the realtime thread wakes up too early.
191  * Lower Limit constraints:
192  * If the early_wakeup time is too small then you again begin to see the affects of scheduling jitter.
193  *
194  * Ideally the early_wakeup time is set to the worst-case timestep - the desired period.
195  * If that number is sufficently small to not starve the non-realtime component of the
196  * system then you should see sub-microsecond jitter in the period.
197  */
198  if (period/10 > 100000ll)
199  /*
200  * Early wake isn't to exceed 100us
201  *
202  * This is a bit arbitrary but if your worst-case timestep exceeds
203  * 100us you are having some problems...
204  */
205  t->wakeup = nano2count(100000ll);
206  else
207  t->wakeup = nano2count(period/10);
208 
209  t->period = nano2count(period);
210  t->next_t = rt_get_time()+t->period;
211 
212  return 0;
213 }

215 void RT::OS::sleepTimestep(RT::OS::Task task) {
216  rtai3_task_t *t = reinterpret_cast<rtai3_task_t *>(task);
217 
218  register RTIME sleep_time;
219 
220  /*
221  * Sleep blocked until early_wakeup nanoseconds before the
222  * next period starts. The actual time is subject to jitter.
223  */
224  rtai_restore_flags(t->eflags);
225  rt_sleep_until(t->next_t-t->wakeup);
226 
227  /*
228  * Busy sleep until it is time for the next period, this
229  * time isn't subject to scheduling jitter.
230  */
231  rtai_save_flags_and_cli(t->eflags);
232  sleep_time = t->next_t-rt_get_time();
233  if (sleep_time > 0)
234  rt_busy_sleep(count2nano(sleep_time));
235 
236  t->next_t += t->period;
237 }

241 int rtai_lxrt_loaded(void) {
242  char buffer[CHUNK_SIZE+1], *tmp;

244

258  } while (size > 0);
259 
260  return -ENOENT;
