42 size_t Fifo::read(void *buffer,size_t n,bool blocking)
43 {
44  // Acquire the data lock
45  if (blocking)
46  pthread_mutex_lock(&mutex);
47  else if (pthread_mutex_trylock(&mutex) != 0)
48  return 0;
49 
50  // Check that enough data is available
51  if (AVAILABLE < n){
52  if (blocking)
53  {
54  do
55  {
56  pthread_cond_wait(&data_available,&mutex);
57  }
58  while (AVAILABLE < n);
59  }
60  else
61  {
62  pthread_mutex_unlock(&mutex);
63  return 0;
64  }
65  }
66  // Copy the data from the fifo
67  if (size-rptr < n)
68  {
69  size_t m = size-rptr;
70  memcpy(buffer,data+rptr,m);
71  memcpy(reinterpret_cast<char *>(buffer)+m,data,n-m);
72  }
73  else
74  memcpy(buffer,data+rptr,n);
75  rptr = (rptr+n)%size;
76 
77  pthread_mutex_unlock(&mutex);
78  return n;
79 }
