57 void rt_switch_warning(int sig, siginfo_t *si, void *context) {
58  const char fmt[] = "\nMode switch (reason: %s). Backtrace:\n";
59  unsigned int reason = si->si_value.sival_int;
60  static char buffer[256];
61  void *bt[200];
62  int nentries;
63 
64  if(reason > SIGDEBUG_WATCHDOG)
65  reason = SIGDEBUG_UNDEFINED;
66 
67  /* Dump a backtrace of the frame which caused the switch to
68 secondary mode: */
69  nentries = snprintf(buffer, sizeof(buffer), fmt, sigdebug_reasons[reason]);
70  nentries = write(STDERR_FILENO, buffer, nentries);
71  nentries = backtrace(bt,sizeof(bt) / sizeof(bt[0]));
72  backtrace_symbols_fd(bt,nentries,fileno(stdout));
73 }
74 #endif

76 int RT::OS::initiate(void) {
77  rt_timer_set_mode(TM_ONESHOT);
78 
79  /**************************************************************************************
80 * On some systems like Fedora Core 4 the memory footprint for rtxi exceeds the mlock *
81 * limit allowed by the linux kernel. As root this limit can be removed with a call *
82 * to setrlimit. However, as a non-root user the call to setrlimit will fail. So if *
83 * you aren't root setrlimit will fail, but that is okay if your system is one that *
84 * doesn't need the extra space. *
85 **************************************************************************************/
86 
87  structrlimit rlim = { RLIM_INFINITY, RLIM_INFINITY };
88  setrlimit(RLIMIT_MEMLOCK,&rlim);
89 
90  if (mlockall(MCL_CURRENT | MCL_FUTURE)) {
91  ERROR_MSG("RTOS:Xenomai::initiate : failed to lock memory.\n");
92  return -EPERM;
93  }
94 
95  pthread_key_create(&is_rt_key,0);
96  init_rt = true;
97 
98  return 0;
99 }

105 int RT::OS::createTask(RT::OS::Task *task,void *(*entry)(void *),void *arg,int prio) {
106  int retval = 0;
107  xenomai_task_t *t = new xenomai_task_t;
108  int priority = 99;
109 
110 #ifdef DEBUG_RT
111  structsigaction sa;
112  sigemptyset(&sa.sa_mask);
113  sa.sa_sigaction = rt_switch_warning;
114  sa.sa_flags = SA_SIGINFO;
115  sigaction(SIGDEBUG, &sa, NULL);
116 #endif
117 
118  // Invert priority, default prio=0 but max priority for xenomai task is 99
119  if ((prio >=0) && (prio <=99))
120  priority -= prio;
121 
122  if ((retval = rt_task_create(&t->task,"RTXI RT Thread",0,priority,T_FPU|T_JOINABLE))) {
123  ERROR_MSG("RT::OS::createTask : failed to create task\n");
124  return retval;
125  }
126 
127  t->period = -1;
128 
129  *task = t;
130  pthread_setspecific(is_rt_key,reinterpret_cast<const void *>(t));
131 
132  if ((retval = rt_task_start(&t->task,reinterpret_cast<void(*)(void *)>(entry),arg))) {
133  ERROR_MSG("RT::OS::createTask : failed to start task\n");
134  return retval;
135  }
136 
137  return 0;
138 }
