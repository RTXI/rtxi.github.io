155 void Connector::Panel::receiveEvent(const Event::Object *event) {
156  if(event->getName() == Event::IO_BLOCK_INSERT_EVENT) {
157  IO::Block *block = reinterpret_cast<IO::Block *>(event->getParam("block"));
158 
159  inputBlock->addItem(QString::fromStdString(block->getName())+QString(" ")+QString::number(block->getID()));
160  outputBlock->addItem(QString::fromStdString(block->getName())+QString(" ")+QString::number(block->getID()));
161  blocks.push_back(block);
162 
163  if(blocks.size() == 1) {
164  buildInputChannelList();
165  buildOutputChannelList();
166  }
167  } else if(event->getName() == Event::IO_BLOCK_REMOVE_EVENT) {
168  IO::Block *block = reinterpret_cast<IO::Block *>(event->getParam("block"));
169 
170  size_t index;
171  for(index = 0;index < blocks.size() && blocks[index] != block;++index);
172  if(index >= blocks.size())
173  return;
174 
175  size_t current0 = inputBlock->currentIndex();
176  size_t current1 = outputBlock->currentIndex();
177 
178  inputBlock->removeItem(index);
179  outputBlock->removeItem(index);
180  blocks.erase(blocks.begin()+index);
181 
182  if(current0 == index) {
183  inputBlock->setCurrentIndex(0);
184  buildInputChannelList();
185  }
186  if(current1 == index) {
187  outputBlock->setCurrentIndex(0);
188  buildOutputChannelList();
189  }
190  } else if(event->getName() == Event::IO_LINK_INSERT_EVENT) {
191  IO::Block *src = reinterpret_cast<IO::Block *>(event->getParam("src"));
192  size_t src_idx = *reinterpret_cast<size_t *>(event->getParam("src_num"));
193  IO::Block *dest = reinterpret_cast<IO::Block *>(event->getParam("dest"));
194  size_t dest_idx = *reinterpret_cast<size_t *>(event->getParam("dest_num"));
195 
196  connectionBox->addItem(QString::number(src->getID())+" "+QString::fromStdString(src->getName())+" : "+
197  QString::number(src_idx)+" "+QString::fromStdString(src->getName(IO::OUTPUT,src_idx))+" ==> "+
198  QString::number(dest->getID())+" "+QString::fromStdString(dest->getName())+" : "+
199  QString::number(dest_idx)+" "+QString::fromStdString(dest->getName(IO::INPUT,dest_idx)));
200  } else if(event->getName() == Event::IO_LINK_REMOVE_EVENT) {
201  IO::Block *src = reinterpret_cast<IO::Block *>(event->getParam("src"));
202  size_t src_idx = *reinterpret_cast<size_t *>(event->getParam("src_num"));
203  IO::Block *dest = reinterpret_cast<IO::Block *>(event->getParam("dest"));
204  size_t dest_idx = *reinterpret_cast<size_t *>(event->getParam("dest_num"));
205 
206  QString link_name = QString::number(src->getID())+" "+QString::fromStdString(src->getName())+" : "+
207  QString::number(src_idx)+" "+QString::fromStdString(src->getName(IO::OUTPUT,src_idx))+" ==> "+
208  QString::number(dest->getID())+" "+QString::fromStdString(dest->getName())+" : "+
209  QString::number(dest_idx)+" "+QString::fromStdString(dest->getName(IO::INPUT,dest_idx));
210 
211  size_t index;
212  for(index=0; index < connectionBox->count() && connectionBox->item(index)->text() != link_name; ++index);
213  if(index >= connectionBox->count())
214  ERROR_MSG("Connector::Panel::receiveEvent : removing non-existant link.\n");
215  else
216  connectionBox->takeItem(index);
217  }
218 }
