253 //
254 // Use createAddressSpace() to create an address space. The registers for the PCI/PXI
255 // board are usually in the BAR1 address space, but the MITE is in the BAR0 address
256 // space.

258 inline tAddressSpace iBus::createAddressSpace(
259  const tBusWindowType windowType,
260  tStatus *callerStatus)
261 {
262  tStatus status;
263  tAddressSpace bar = 0x0;
264 
265  status = kStatusSuccess;
266 
267  switch (windowType)
268  {
269  case kPCI_BAR0:
270  bar = _bar[0];
271  break;
272 
273  case kPCI_BAR1:
274  bar = _bar[1];
275  break;
276 
277  case kPCI_BAR2:
278  bar = _bar[2];
279  break;
280 
281  case kPCI_BAR3:
282  bar = _bar[3];
283  break;
284 
285  case kPCI_BAR4:
286  bar = _bar[4];
287  break;
288 
289  case kPCI_BAR5:
290  bar = _bar[5];
291  break;
292 
293  default:
294  status = kStatusBadWindowType;
295  break;
296  }
297 
298  if (callerStatus!=NULL) *callerStatus = status;
