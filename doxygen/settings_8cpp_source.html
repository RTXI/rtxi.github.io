203 int Settings::Manager::load(const std::string &filename) {
204  QFile file(QString::fromStdString(filename));
205  if (!file.open(QIODevice::ReadOnly)) {
206  ERROR_MSG("Settings::Manager::load : failed to open %s for reading\n", filename.c_str());
207  return -EPERM;
208  }
209 
210  QDomDocument doc;
211  QString errorMsg;
212  int errorLine, errorColumn;
213 
214  if (!doc.setContent(&file,false,&errorMsg,&errorLine,&errorColumn)) {
215  ERROR_MSG("Settings::Manager::load : %s:%d:%d: %s\n", filename.c_str(), errorLine, errorColumn, errorMsg.toStdString().c_str());
216  return -EINVAL;
217  }
218 
219  QDomElement e1 = doc.documentElement();
220 
221  if (e1.tagName() != "RTXI" || e1.attribute("class") != "settings") {
222  ERROR_MSG("Settings::Manager::load : invalid document element\n");
223  return -EINVAL;
224  }
225 
226  // Return RTXI to a startup like state.
227  Plugin::Manager::getInstance()->unloadAll();
228  MainWindow::getInstance()->clearFileMenu();
229  MainWindow::getInstance()->clearModuleMenu();
230 
231  // Reading in the period for the system
232  long long period = RT::System::getInstance()->getPeriod();
233  RT::System::getInstance()->setPeriod(1000000); // ns equivalent to 1ms (1kHz)
234 
235  Object::State s;
236  Plugin::Object *plugin;
237  std::list<defer_t> deferList;
238  for (QDomElement e2 = e1.firstChild().toElement(); !e2.isNull(); e2 = e2.nextSibling().toElement()) {
239  if (e2.tagName() != "OBJECT" || e2.attribute("component") == QString::null) continue;
240 
241  s.xml(e2);
242  // Load plugin info
243  if (e2.attribute("component") == "plugin") {
244  if ((plugin = Plugin::Manager::getInstance()->load(e2.attribute("library")))) {
245  defer_t defer = { plugin, s };
246  deferList.push_back(defer);
247  plugin->load(s);
248  }
249  }
250  // Load RT info
251  else if (e2.attribute("component") == "rt") {
252  period = strtoll(s.loadString("Period").c_str(),0,10);
253  // Legacy case, period is stored as a double in scientific notation
254  if (period < 1000)
255  period = s.loadDouble("Period");
256  } // Load IO info
257  else if (e2.attribute("component") == "io") {
258  defer_t defer = { IO::Connector::getInstance(), s };
259  deferList.push_back(defer);
260  }
261  }
262  for (std::list<defer_t>::iterator i = deferList.begin(),end = deferList.end(); i != end; ++i)
263  i->object->deferred(i->s);
264 
265  if (period)
266  RT::System::getInstance()->setPeriod(period);
267 
268  // create QSettings
269  QSettings userprefs;
270  userprefs.setPath(QSettings::NativeFormat, QSettings::SystemScope, "/usr/local/share/rtxi/");
271 
272  int oldestsetting = userprefs.value("/recentSettingsList/start").toInt();
273  int num_settings = userprefs.value("/recentSettingsList/num").toInt();
274 
275  userprefs.beginGroup("/recentSettingsList");
276  QStringList entries = userprefs.childKeys();
277  userprefs.endGroup();
278  int numRecentFiles = entries.size();
279  QString listsetting;
280  bool doesnotexist = true;
281 
282  for (int i = 0; i < numRecentFiles; ++i) {
283  listsetting = userprefs.value("/recentSettingsList/" + entries[i]).toString();
284  if (QString::fromStdString(filename) == listsetting)
285  doesnotexist = false;
286  }
287 
288  if (filename == "/etc/rtxi.conf")
289  doesnotexist = false;
290 
291  if (doesnotexist) {
292  if (num_settings == 10) {
293  userprefs.setValue("/recentSettingsList/" + QString::number(oldestsetting), QVariant(QString::fromStdString(filename)));
294  oldestsetting++;
295  if (oldestsetting == 10)
296  oldestsetting = 0;
297  userprefs.setValue("/recentSettingsList/start", oldestsetting);
298  }
299  else {
300  userprefs.setValue("/recentSettingsList/" + QString::number(num_settings++), QVariant(QString::fromStdString(filename)));
301  userprefs.setValue("/recentSettingsList/num", num_settings);
302  }
303  }
304 
305  return 0;
306 }

397 void Settings::Manager::acquireID(Settings::Object *object,Settings::Object::ID id) {
398  Mutex::Locker lock(&mutex);
399 
400  if (id != Object::INVALID) {
401  if (objectMap.find(id) == objectMap.end()) {
402  objectMap[id] = object;
403  object->id = id;
404  return;
405  } else
406  DEBUG_MSG("Settings::Manager::acquireID : requested ID in use\n");
407  }
408 
409  /***********************************************************
410  * Traverse the ID space to find an available ID. *
411  * It ain't pretty, O(n^2) if my math is correct... *
412  ***********************************************************/
413 
414  while (objectMap.find(currentID) != objectMap.end() && currentID != Object::INVALID) ++currentID;
415 
416  /*****************************************************************************
417  * It is possible that there are no more available IDs, and in that case *
418  * the insertion fails and the object is assigned an invalid ID. *
419  * In practice this shouldn't ever happen, because on the x86 we have 2^32 *
420  * IDs, so we would probably run out of memory before hitting this limit *
421  *****************************************************************************/
422 
423  if (currentID == Object::INVALID) {
424  ERROR_MSG("Settings::Manager::acquireID : maximum number of settings objects loaded.\n");
425  object->id = Object::INVALID;
426  } else {
427  object->id = currentID;
428  objectMap[currentID++] = object;
429  }
430 }
