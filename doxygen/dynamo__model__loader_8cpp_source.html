96 void DynamoModelLoader::load(char* srcpath) {
97  int status;
98  char *path;
99 
100  path = canonicalize_file_name(srcpath);
101  QString model_name = QString(path);
102  model_name.remove(MODEL_SOURCE_SUFFIX);
103 
104  if (model_name.isNull() || model_name.isEmpty())
105  return;
106 
107  if (model_makefile_path.isNull() || model_makefile_path.isEmpty()) {
108  ERROR_MSG("invalid model makefile name\n");
109  return;
110  }
111 
112  QString cmd = QString("cd %1; %2 %3 %4%5") .arg(getcwd(NULL, 0)) .arg(
113  MAKE_CMD_PREFIX) .arg(model_makefile_path) .arg(model_name) .arg(
114  MODEL_SUFFIX);
115 
116  QString module_name = QString("%1%2") .arg(model_name) .arg(MODEL_SUFFIX);
117 
118  DEBUG_MSG("about to compile model with command %s\n", cmd.toLatin1().constData());
119  status = CmdLine::getInstance()->execute(cmd.toStdString());
120 
121  if (status != 0) {
122  ERROR_MSG("build process error\n");
123  return;
124  }
125 
126  Plugin::Manager::getInstance()->load(module_name.toLatin1());
127 
128  if (path != NULL)
129  free(path);
130 }
