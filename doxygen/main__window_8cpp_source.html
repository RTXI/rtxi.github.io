132 void MainWindow::createUtilMenu() {
133  utilMenu = menuBar()->addMenu(tr("&Utilities"));
134  filtersSubMenu = utilMenu->addMenu(tr("&Filters"));
135  signalsSubMenu = utilMenu->addMenu(tr("&Signals"));
136  utilitiesSubMenu = utilMenu->addMenu(tr("&Utilities"));
137  connect(utilMenu, SIGNAL(triggered(QAction*)), this, SLOT(utilitiesMenuActivated(QAction*)));
138 
139  QDir libsDir("/usr/local/lib/rtxi/");
140  if(!libsDir.exists())
141  return;
142 
143  libsDir.setNameFilters(QStringList("*.so"));
144  for(size_t i = 0; i < libsDir.entryList().size(); i++)
145  {
146  utilItem = new QAction(libsDir.entryList().at(i), this);
147  if(libsDir.entryList().at(i).contains("analysis"))
148  utilitiesSubMenu->addAction(utilItem);
149  else if(libsDir.entryList().at(i).contains("sync"))
150  utilitiesSubMenu->addAction(utilItem);
151  else if(libsDir.entryList().at(i).contains("mimic"))
152  utilitiesSubMenu->addAction(utilItem);
153  else if(libsDir.entryList().at(i).contains("iir"))
154  filtersSubMenu->addAction(utilItem);
155  else if(libsDir.entryList().at(i).contains("fir"))
156  filtersSubMenu->addAction(utilItem);
157  else if(libsDir.entryList().at(i).contains("signal"))
158  signalsSubMenu->addAction(utilItem);
159  else if(libsDir.entryList().at(i).contains("noise"))
160  signalsSubMenu->addAction(utilItem);
161  else if(libsDir.entryList().at(i).contains("ttl"))
162  signalsSubMenu->addAction(utilItem);
163  else if(libsDir.entryList().at(i).contains("maker"))
164  signalsSubMenu->addAction(utilItem);
165  }
166 }

280  QSettings userprefs;

282  QString settingsDir = userprefs.value("/dirs/setfiles", getenv("HOME")).toString();
