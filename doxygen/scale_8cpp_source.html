212 {

222  bar1 = bus->createAddressSpace(kPCI_BAR1);
223 
224  // ---- Open EEPROM
225 
226  _iodwbsrInitial = bar0.read32 (0xC0);
227  bar0.write32 (0xC0, 0x00);
228 
229  _iowbsr1Initial = bar0.read32 (0xC4);

237  bar0.write32 (0x30, 0xF);

242  {
243  eepromBuffer[i] = bar1.read8(i + offset);
244  }
245

250  bar0.write32 (0xF4, _iowcr1Initial);
