37 RealTimeMath::fastEXP(double x)
38 {
39  if (x > 88.5 ||
40  x < -87) { // Overflow prevention due to powFast library's use of float
41 
42  // Close to 0 shortcut: If x < -746, math.h exp(x) function returns 0 due to
43  // truncation
44  if (x < -746)
45  return 0;
46 
47  // Infinity shortcut: If x > 710, math.h exp(x) returns infinity
48  if (x > 710) {
49  return INFINITY;
50  }
51 
52  // Standard overflow prevention
53  // Due to powFast limitations from using floats, if x is too large or
54  // small, problem is broken into smaller pieces
55  // e.g. fastEXP(100) ==> e^180 = e^60 * e^60 * e^60
56  powCount = 1;
57  powAns = 1;
58  powExp = x;
59 
60  while (powExp > 88.5 || powExp < -87) {
61  powCount++;
62  powExp = x / powCount;
63  }
64 
65  for (n = 1; n <= powCount; n++) {
66  powAns = powAns * powFast->e(powExp);
67  }
68 
69  return powAns;
70  } // end overflow prevention
71 
72  else // call powFast e^x function
73  return powFast->e(x);
74 }
