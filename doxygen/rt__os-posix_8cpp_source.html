71 static void *bounce(void *bounce_info) {
72  posix_bounce_info_t *info = reinterpret_cast<posix_bounce_info_t *>(bounce_info);
73 
74  posix_task_t *t = info->t;
75  void *(*entry)(void *) = info->entry;
76  void *arg = info->arg;
77 
78  t->period = -1;
79  t->next_t = -1;
80  t->thread = pthread_self();
81 
82  pthread_setspecific(is_rt_key,reinterpret_cast<const void *>(t));
83 
84  sem_post(&info->sem);
85  return entry(arg);
86 }

142 void RT::OS::sleepTimestep(RT::OS::Task task) {
