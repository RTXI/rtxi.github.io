146  double work;

197 double
198 RemezAlgorithm::ComputeRemezAmplitudeResponse(int init_flag, double contin_freq)
199 {
200  static int i, j, k, sign;
201  static double freq, denom, numer, alpha, delta;
202  static double absDelta, xCont, term;
203  double aa;
204 
205  if (init_flag) {
206  X_ = new double[Num_Approx_Funcs + 1];
207  Beta_ = new double[Num_Approx_Funcs + 1];
208  Gamma_ = new double[Num_Approx_Funcs + 1];
209  for (j = 0; j <= Num_Approx_Funcs; j++) {
210  freq = GetFrequency(Ext_Freq[j]);
211  X_[j] = cos(TWO_PI * freq);
212  }
213 
214  // compute delta
215  denom = 0.0;
216  numer = 0.0;
217  sign = -1;
218  for (k = 0; k <= Num_Approx_Funcs; k++) {
219  sign = -sign;
220  alpha = 1.0;
221  for (i = 0; i <= (Num_Approx_Funcs - 1); i++) {
222  if (i == k)
223  continue;
224  alpha = alpha / (X_[k] - X_[i]);
225  }
226  Beta_[k] = alpha;
227  if (k != Num_Approx_Funcs)
228  alpha = alpha / (X_[k] - X_[Num_Approx_Funcs]);
229  freq = GetFrequency(Ext_Freq[k]);
230  numer = numer + alpha * DesiredResponse(freq);
231 
232  denom = denom + sign * (alpha / WeightFunction(freq));
233  } // end of loop over k
234 
235  delta = numer / denom;
236  absDelta = fabs(delta);
237 
238  sign = -1;
239  for (k = 0; k <= Num_Approx_Funcs - 1; k++) {
240  sign = -sign;
241  freq = GetFrequency(Ext_Freq[k]);
242  Gamma_[k] = DesiredResponse(freq) - sign * delta / WeightFunction(freq);
243  }
244  } // end of if(init_flag)
245  else {
246  xCont = cos(TWO_PI * contin_freq);
247  numer = 0.0;
248  denom = 0.0;
249  for (k = 0; k < Num_Approx_Funcs; k++) {
250  term = xCont - X_[k];
251  if (fabs(term) < 1.0e-7) {
252  aa = Gamma_[k];
253  goto done;
254  } else {
255  term = Beta_[k] / (xCont - X_[k]);
256  denom += term;
257  numer += Gamma_[k] * term;
258  }
259  }
260  aa = numer / denom;
261  }
262 done:
263  return (aa);
264 }
