44 
45  void *handle = dlopen(library.toStdString().c_str(),RTLD_GLOBAL|RTLD_NOW);
46  if (!handle) {
47  std::string plugin_dir = std::string(EXEC_PREFIX) + std::string("/lib/rtxi/");
48  handle = dlopen((plugin_dir+library.toStdString()).c_str(),RTLD_GLOBAL|RTLD_NOW);
49  }
50  if (!handle) {
51  ERROR_MSG("Plugin::load : failed to load %s: %s\n",library.toStdString().c_str(),dlerror());

57 * But what the hell do they know? It is probably safe here... *
58 *********************************************************************************/
