204 void DefaultGUIModel::refresh(void) {
205  for (std::map<QString, param_t>::iterator i = parameter.begin(); i!= parameter.end(); ++i) {
206  if (i->second.type & (STATE | EVENT)) {
207  i->second.edit->setText(QString::number(getValue(i->second.type, i->second.index)));
208  palette.setBrush(i->second.edit->foregroundRole(), Qt::darkGray);
209  i->second.edit->setPalette(palette);
210  } else if ((i->second.type & PARAMETER) && !i->second.edit->isModified()
211  && i->second.edit->text() != *i->second.str_value) {
212  i->second.edit->setText(*i->second.str_value);
213  } else if ((i->second.type & COMMENT) && !i->second.edit->isModified()
214  && i->second.edit->text() != QString::fromStdString(getValueString(COMMENT, i->second.index))) {
215  i->second.edit->setText(QString::fromStdString(getValueString(COMMENT, i->second.index)));
216  }
217  }
218  pauseButton->setChecked(!getActive());
219 }
