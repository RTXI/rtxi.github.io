306 {

308  return -EINVAL;
309

397 void AnalogyDevice::read(void) {
398  lsampl_t sample;
399  analog_channel_t *channel;
400  int ref = 0;
401  int size = 0;
402  int err = -1;
403  a4l_chinfo_t *chinfo;
404 
405  for(size_t i=0;i < subdevice[AI].count;++i)
406  if(subdevice[AI].chan[i].active) {
407  channel = &subdevice[AI].chan[i].analog;
408 
409  // Get analogy reference
410  switch (channel->reference) {
411  case 0: ref = A4L_CHAN_AREF_GROUND; break;
412  case 1: ref = A4L_CHAN_AREF_COMMON; break;
413  case 2: ref = A4L_CHAN_AREF_DIFF; break;
414  case 3: ref = A4L_CHAN_AREF_OTHER; break;
415  }
416 
417  // Get channel size
418  a4l_get_chinfo(&dsc, subdevice[AI].id, i, &chinfo);
419  size = a4l_sizeof_chan(chinfo);
420 
421  // Read 1 data sample via synchronous acq
422  sample = 0;
423  err = a4l_sync_read(&dsc, subdevice[AI].id, PACK(i,channel->range,ref), 0, &sample, size);
424 
425  // Read via asynchronous acq
426  //err = a4l_async_read(&dsc, &sample, size, 1);
427 
428  output(i) = channel->gain*channel->conv*(sample-channel->offset)+channel->zerooffset;
429  }
430 
431  unsigned int data = 0, mask = 0xff;
432  size_t offset = getChannelCount(AI);
433 
434  // Read all data and output it according to channel activity
435  a4l_sync_dio(&dsc, subdevice[DIO].id, &mask, &data);
436  for(size_t i=0;i < subdevice[DIO].count;++i)
437  if(subdevice[DIO].chan[i].active && subdevice[DIO].chan[i].digital.direction == DAQ::INPUT) {
438  mask = (1<<i);
439  output(i+offset) = (data & mask);
440  }
441 }

443 void AnalogyDevice::write(void) {
444  {
445  double value;
446  lsampl_t sample;
447  analog_channel_t *channel;
448  int ref = 0;
449  int size = 0;
450  int err = -1;
451  a4l_chinfo_t *chinfo;
452 
453  for(size_t i=0;i < subdevice[AO].count;++i)
454  if(subdevice[AO].chan[i].active) {
455  channel = &subdevice[AO].chan[i].analog;
456  value = round(channel->gain*channel->conv*(input(i)-channel->zerooffset)+channel->offset);
457 
458  /*
459  * Prevent wrap around in the data units.
460  */
461  if(value > channel->maxdata)
462  value = channel->maxdata;
463  else if(value < 0.0)
464  value = 0.0;
465  sample = static_cast<lsampl_t>(value);
466 
467  // Get anaolgy reference
468  switch (channel->reference) {
469  case 0: ref = A4L_CHAN_AREF_GROUND; break;
470  case 1: ref = A4L_CHAN_AREF_COMMON; break;
471  case 2: ref = A4L_CHAN_AREF_DIFF; break;
472  case 3: ref = A4L_CHAN_AREF_OTHER; break;
473  }
474  // Get channel size
475  a4l_get_chinfo(&dsc, subdevice[AO].id, i, &chinfo);
476  size = a4l_sizeof_chan(chinfo);
477  // Write sample
478  err = a4l_sync_write(&dsc, subdevice[AO].id, PACK(i,channel->range,ref),
479  0, &sample, size);
480  }
481  }
482 
483  {
484  size_t offset = getChannelCount(AO);
485  int value;
486  int data = 0, mask = 0;
487 
488  for(size_t i=0;i < subdevice[DIO].count;++i) {
489  value = input(i+offset) != 0.0;
490  if(subdevice[DIO].chan[i].active && subdevice[DIO].chan[i].digital.direction == DAQ::OUTPUT && subdevice[DIO].chan[i].digital.previous_value != value) {
491  subdevice[DIO].chan[i].digital.previous_value = value;
492  data ^= (1<<i); // Toggle the i-th bit
493  mask |= (1<<i); // Set i-th bit for modification
494  }
495  }
496  // Write the data according to the mask
497  if (mask) {
498  a4l_sync_dio(&dsc, subdevice[DIO].id, &mask, &data);
499  }
