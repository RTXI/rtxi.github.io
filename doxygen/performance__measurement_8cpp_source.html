117 void PerformanceMeasurement::Panel::read(void) {
118  long long now = RT::OS::getTime();
119 
120  switch (state) {
121  case EXEC:
122  if (maxTimestep < now - lastRead)
123  maxTimestep = now - lastRead;
124  timestep = now - lastRead;
125  timestepStat.push(timestep);
126  break;
127  case INIT2:
128  timestep = maxTimestep = now - lastRead;
129  timestepStat.push(timestep);
130  state = EXEC;
131  break;
132  case INIT1:
133  state = INIT2;
134  }
135  lastRead = now;
136  jitter = timestepStat.std();
137 }

139 void PerformanceMeasurement::Panel::write(void) {
140  long long now = RT::OS::getTime();
141 
142  switch (state) {
143  case EXEC:
144  if (maxDuration < now - lastRead)
145  {
146  maxDuration = now - lastRead;
147  resetMaxTimer->start(10000);
148  }
149  duration = now - lastRead;
150  break;
151  case INIT2:
152  duration = maxDuration = now - lastRead;
153  break;
154  default:
155  ERROR_MSG("PerformanceMeasurement::Panel::write : invalid state\n");
156  }
157 }
