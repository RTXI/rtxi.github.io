125 void PerformanceMeasurement::Panel::read(void) {
126  long long now = RT::OS::getTime();
127 
128  switch (state) {
129  case EXEC:
130  if (maxTimestep < now - lastRead)
131  maxTimestep = now - lastRead;
132  timestep = now - lastRead;
133  timestepStat.push(timestep);
134  break;
135  case INIT2:
136  timestep = maxTimestep = now - lastRead;
137  timestepStat.push(timestep);
138  state = EXEC;
139  break;
140  case INIT1:
141  state = INIT2;
142  }
143  lastRead = now;
144  jitter = timestepStat.std();
145 }

147 void PerformanceMeasurement::Panel::write(void) {
148  long long now = RT::OS::getTime();
149 
150  switch (state) {
151  case EXEC:
152  if (maxDuration < now - lastRead)
153  maxDuration = now - lastRead;
154  duration = now - lastRead;
155  break;
156  case INIT2:
157  duration = maxDuration = now - lastRead;
158  break;
159  default:
160  ERROR_MSG("PerformanceMeasurement::Panel::write : invalid state\n");
161  }
162 }
