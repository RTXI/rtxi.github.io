64 Workspace::Instance::Instance(std::string name,Workspace::variable_t *d,size_t n)
65  : IO::Block(name,d,n) {
66  size_t count[] = { 0, 0, 0, 0 };
67  for (size_t i=0; i<n; ++i)
68  switch (d[i].flags & (PARAMETER|STATE|EVENT|COMMENT)) {
69  case PARAMETER:
70  count[0]++;
71  break;
72  case STATE:
73  count[1]++;
74  break;
75  case EVENT:
76  count[2]++;
77  break;
78  case COMMENT:
79  count[3]++;
80  break;
81  }
82 
83  parameter = std::vector<var_t>(count[0]);
84  state = std::vector<var_t>(count[1]);
85  event = std::vector<var_t>(count[2]);
86  comment = std::vector<comment_t>(count[3]);
87 
88  size_t i[] = { 0, 0, 0, 0 };
89  for (size_t j=0; j<n; ++j) {
90  switch (d[j].flags & (PARAMETER|STATE|EVENT|COMMENT)) {
91  case PARAMETER:
92  parameter[i[0]].name = d[j].name;
93  parameter[i[0]].description = d[j].description;
94  parameter[i[0]].data = new double;
95  i[0]++;
96  break;
97  case STATE:
98  state[i[1]].name = d[j].name;
99  state[i[1]].description = d[j].description;
100  state[i[1]].data = 0;
101  i[1]++;
102  break;
103  case EVENT:
104  event[i[2]].name = d[j].name;
105  event[i[2]].description = d[j].description;
106  event[i[2]].data = 0;
107  i[2]++;
108  break;
109  case COMMENT:
110  comment[i[3]].name = d[j].name;
111  comment[i[3]].description = d[j].description;
112  comment[i[3]].comment = "";
113  i[3]++;
114  break;
115  }
116  }

171 double Workspace::Instance::getValue(IO::flags_t type,size_t n)const{
172  if (type & INPUT)
173  return input(n);
174  if (type & OUTPUT)
175  return output(n);
176  if (type & PARAMETER && n < parameter.size() && parameter[n].data)
177  return *parameter[n].data;
178  if (type & STATE && n < state.size() && state[n].data)
179  return *state[n].data;
180  if (type & EVENT && n < event.size() && event[n].data)
181  return *event[n].data;
182  if (type & COMMENT && n < comment.size()) {
183  std::istringstream sstr(comment[n].comment);
184  double value;
185  sstr >> value;
186  return value;
187  }
188  return 0.0;
