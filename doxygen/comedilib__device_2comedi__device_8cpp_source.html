370 int ComediDevice::setAnalogCalibration(type_t type,index_t channel) {
371  if(!analog_exists(type,channel))
372  return -EINVAL;
373 
374  analog_channel_t *chanPtr = &subdevice[type].chan[channel].analog;
375  comedi_polynomial_t polynomial;
376  int retval = -1;
377 
378  if( calibration != NULL ) { // Board was soft calibrated
379  DEBUG_MSG("ComediDevice::setAnalogCalibration - Soft calibration used for [subd,channel] [%i,%i]\n",
380  subdevice[type].id,channel);
381 
382  if( type == AI )
383  retval = comedi_get_softcal_converter(subdevice[type].id,channel,chanPtr->range,
384  COMEDI_TO_PHYSICAL,calibration,&polynomial);
385  else if( type == AO )
386  retval = comedi_get_softcal_converter(subdevice[type].id,channel,chanPtr->range,
387  COMEDI_FROM_PHYSICAL,calibration,&polynomial);
388  else
389  ERROR_MSG("ComediDevice::setAnalogCalibration : invalid type\n");
390 
391  if( retval < 0 ) {
392  ERROR_MSG("ComediDevice::setAnalogCalibration : unable to retrieve softcal converter\n");
393  chanPtr->calibrated = false;
394  }
395  else { // Calibration retrieval successful
396  chanPtr->calibrated = true;
397  chanPtr->order = polynomial.order;
398  chanPtr->expansionOrigin = polynomial.expansion_origin;
399 
400  for( int i = 0; i < 4; i++ ) // Max 4 coefficients
401  chanPtr->coefficients[i] = polynomial.coefficients[i];
402  }
403  }
404  else { // Check if board is hard calibrated
405  DEBUG_MSG("ComediDevice::setAnalogCalibration - Soft calibration used for [subd,channel] [%i,%i]\n",
406  subdevice[type].id,channel);
407 
408  if( type == AI )
409  retval = comedi_get_hardcal_converter(device,subdevice[type].id,channel,chanPtr->range,
410  COMEDI_TO_PHYSICAL,&polynomial);
411  else if( type == AO )
412  retval = comedi_get_hardcal_converter(device,subdevice[type].id,channel,chanPtr->range,
413  COMEDI_FROM_PHYSICAL,&polynomial);
414  else
415  ERROR_MSG("ComediDevice::setAnalogCalibration : invalid type\n");
416 
417  if( retval < 0 ) {
418  ERROR_MSG("ComediDevice::setAnalogCalibration : unable to retrieve calibration, no calibration is being used\n");
419  chanPtr->calibrated = false;
420  }
421  else { // Calibration retrieval successful
422  chanPtr->calibrated = true;
423  chanPtr->order = polynomial.order;
424  chanPtr->expansionOrigin = polynomial.expansion_origin;
425 
426  for( int i = 0; i < 4; i++ ) // Max 4 coefficients
427  chanPtr->coefficients[i] = polynomial.coefficients[i];
428  }
429  }
430 
431  return 0;
432 }

455 void ComediDevice::read(void)
456 {
457  lsampl_t sample;
458  analog_channel_t *channel;
459 
460  for(size_t i=0;i < subdevice[AI].count;++i)
461  if(subdevice[AI].chan[i].active) {
462  channel = &subdevice[AI].chan[i].analog;
463  comedi_data_read(device,subdevice[AI].id,i,channel->range,channel->reference,&sample);
464  if( channel->calibrationActive ) { // Use polynomial fitting if channel is active
465  double value = 0.;
466  double term = 1.;
467 
468  for(unsigned j = 0; j <= channel->order; ++j) {
469  value += channel->coefficients[j] * term;
470  term *= sample - channel->expansionOrigin;
471  }
472  output(i) = channel->gain * value + channel->zerooffset;
473  }
474  else // No calibration is used
475  output(i) = channel->gain * channel->conv * (sample-channel->offset) + channel->zerooffset;
476  }
477 
478  unsigned int data;
479  size_t offset = getChannelCount(AI);
480 
481  for(size_t i=0;i < subdevice[DIO].count;++i)
482  if(subdevice[DIO].chan[i].active && subdevice[DIO].chan[i].digital.direction == DAQ::INPUT) {
483  comedi_dio_read(device,subdevice[DIO].id,i,&data);
484  output(i+offset) = data;
485  }

488 void ComediDevice::write(void)
489 {
490  {
491  double value;
492  lsampl_t sample;
493  analog_channel_t *channel;
494 
495  for(size_t i=0;i < subdevice[AO].count;++i)
496  if(subdevice[AO].chan[i].active) {
497  channel = &subdevice[AO].chan[i].analog;
498 
499  if( channel->calibrationActive ) { // Use polynomial fitting if channel calibration is active
500  double in = channel->gain * ( input(i) + channel->zerooffset ); // Physical value
501 
502  value = 0.;
503  double term = 1.;
504 
505  for(unsigned j = 0; j <= channel->order; ++j) {
506  value += channel->coefficients[j] * term;
507  term *= in - channel->expansionOrigin;
508  }
509 
510  value = round( value );
511  }
512  else // No calibration is used
513  value = round(channel->gain*channel->conv*(input(i)+channel->zerooffset)+channel->offset);
514 
515  /*
516  * Prevent wrap around in the data units.
517  */
518  if(value > channel->maxdata)
519  value = channel->maxdata;
520  else if(value < 0.0)
521  value = 0.0;
522 
523  sample = static_cast<sampl_t>(value);
524 
525  comedi_data_write(device,subdevice[AO].id,i,channel->range,channel->reference,sample);
526  }
527  }
528 
529  {
530  size_t offset = getChannelCount(AO);
531  int value;
532 
533  for(size_t i=0;i < subdevice[DIO].count;++i) {
534  value = input(i+offset) != 0.0;
535  if(subdevice[DIO].chan[i].active && subdevice[DIO].chan[i].digital.direction == DAQ::OUTPUT && subdevice[DIO].chan[i].digital.previous_value != value) {
536  subdevice[DIO].chan[i].digital.previous_value = value;
537  comedi_dio_write(device,subdevice[DIO].id,i,value);
538  }
539  }
540  }
