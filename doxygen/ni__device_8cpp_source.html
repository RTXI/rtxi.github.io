38  int NIDevice::AIConfig::callback(void) {

40  subd->chan[channel].analog.active = active;
41 
42  // ---- AI Reset ----
43  //
44 
45  configureTimebase (board);
46  pllReset (board);
47  analogTriggerReset (board);
48 
49  aiReset (board);
50  if(info->select_active_high)
51  aiPersonalize (board, tMSeries::tAI_Output_Control::kAI_CONVERT_Output_SelectActive_High);
52  else
53  aiPersonalize (board, tMSeries::tAI_Output_Control::kAI_CONVERT_Output_SelectActive_Low);
54  aiClearFifo (board);
55 
56  if(info->adc_reset)

58 
59  // ---- End of AI Reset ----
60 
61  // ---- Start AI task ----
62 
63  aiDisarm (board);
64  aiClearConfigurationMemory (board);
65 
66  // fill configuration FIFO
67  // Note: It is not necessary for the channel numbers to be ordered
68  subd->active = 0;
69  for(size_t i=0;i<subd->count;++i)
70  if(subd->chan[i].analog.active)
71  ++subd->active;
72  size_t n = subd->active;
73  for (size_t i=0;i<subd->count && n;++i)
74  if(subd->chan[i].analog.active)
75  aiConfigureChannel (board,
76  i, // channel number
77  info->ai_gains[subd->chan[i].analog.range].gain_idx, // gain -- check ai.h for allowed values
78  tMSeries::tAI_Config_FIFO_Data::kAI_Config_PolarityBipolar,
79  tMSeries::tAI_Config_FIFO_Data::kAI_Config_Channel_TypeDifferential,
80  (--n)?kFalse:kTrue); // last channel?
81

83 
84  aiEnvironmentalize (board);
85 
86  aiHardwareGating (board);
87 
88  aiTrigger (board,
89  tMSeries::tAI_Trigger_Select::kAI_START1_SelectPulse,
90  tMSeries::tAI_Trigger_Select::kAI_START1_PolarityRising_Edge,
91  tMSeries::tAI_Trigger_Select::kAI_START2_SelectPulse,
92  tMSeries::tAI_Trigger_Select::kAI_START2_PolarityRising_Edge);
93 
94  aiSampleStop (board,
95  (subd->active > 1)?kTrue:kFalse); // multi channel?
96 
97  aiNumberOfSamples (board,
98  1, // posttrigger samples
99  0, // pretrigger samples
100  kTrue); // continuous?
101 
102  aiSampleStart (board,

104  0,
105  tMSeries::tAI_START_STOP_Select::kAI_START_SelectPulse,
106  tMSeries::tAI_START_STOP_Select::kAI_START_PolarityRising_Edge);
107 
108  aiConvert (board,

110  info->delay_divisor, // convert delay divisor
111  kFalse); // external sample clock?
112 
113  aiClearFifo (board);
114 
115  // check scale.h for scaling index values

117  aiGetScalingCoefficients(eepromMemory,info->ai_gains[subd->chan[i].analog.range].scale_idx,0,i,&subd->chan[i].analog.scale);

458 void NIDevice::read(void) {
459  size_t count;
460 
461  if(subdevice[AI].active) {
462  aiStartOnDemand (board);
463  while (board->Joint_Status_2.readAI_Scan_In_Progress_St()) {}
464 
465  i32 value;
466  f32 scaled;
467 
468  count = 0;
469  for(size_t i=0;i<getChannelCount(AI);++i) {
470  if(getChannelActive(AI,i)) {
471  value = board->AI_FIFO_Data.readRegister ();
472  aiPolynomialScaler (&value,&scaled,&subdevice[AI].chan[i].analog.scale);
473  output(i) = subdevice[AI].chan[i].analog.gain*scaled+subdevice[AI].chan[i].analog.zerooffset;
474 
475  if(++count >= subdevice[AI].active) break;
476  }
477  }
478  }
479 
480  if(subdevice[DIO].active) {
481  size_t offset;
482  unsigned int data = board->Static_Digital_Input.readRegister();
483 
484  count = 0;
485  offset = getChannelCount(AI);
486  for(size_t i=0;count < getChannelCount(DIO);++i)
487  if(getChannelActive(DIO,i)) {
488  if(subdevice[DIO].chan[i].digital.direction == DAQ::INPUT) {
489  if(data & 1<<i)
490  output(i+offset) = 5.0;
491  else
492  output(i+offset) = 0.0;
493  } else {
494  if(input(i+getChannelCount(AO)) > 2.0)
495  output(i+offset) = 5.0;
496  else
497  output(i+offset) = 0.0;
498  }
499 
500  if(++count >= subdevice[DIO].active) break;
501  }
502  }
503 }

505 void NIDevice::write(void) {
506  size_t count;
507 
508  if(subdevice[AO].active) {
509  i32 value;
510  f32 voltage;
511 
512  count = 0;
513  for(size_t i=0;i<getChannelCount(AO);++i)
514  if(getChannelActive(AO,i)) {
515  voltage = (input(i)-subdevice[AO].chan[i].analog.zerooffset)*subdevice[AO].chan[i].analog.gain;
516  aoLinearScaler(&value,&voltage,&subdevice[AO].chan[i].analog.scale);
517  board->DAC_Direct_Data[i].writeRegister(value);
518 
519  if(++count >= subdevice[AO].active) break;
520  }
521  }
522 
523  if(subdevice[DIO].active) {
524  size_t offset;
525  unsigned int data = 0;
526 
527  count = 0;
528  offset = getChannelCount(AO);
529  for(size_t i=0;count < getChannelCount(DIO);++i)
530  if(subdevice[DIO].chan[i].digital.active) {
531  ++count;
532  if(input(i+offset) > 2.0)
533  data |= 1<<i;
534  }
535  board->Static_Digital_Output.writeRegister(data);
536  }
